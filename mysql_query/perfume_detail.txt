CREATE TABLE `perfumedetail`.`detail` (

`pname` VARCHAR(30) NOT NULL , 
`pbrand` VARCHAR(30) NOT NULL , 
`pimage` VARCHAR(200) NOT NULL , 
`pexp` VARCHAR(100) NOT NULL , 
`ptag1` VARCHAR(30) NOT NULL , 
`ptag2` VARCHAR(30) NULL ,

) ENGINE = InnoDB;
